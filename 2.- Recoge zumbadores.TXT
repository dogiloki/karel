class Robot{
    Robot(){
        while(frontIsClear || facingNorth || facingSouth){
            while(nextToABeeper){
                pickbeeper();
            }
            if(frontIsClear){
                move();
            }else{
                while(!facingNorth){
                    turnleft();
                }
                if(rightIsClear){
                    while(!facingEast){
                        turnleft();
                    }
                    if(frontIsClear){
                        move();
                    }
                    while(!facingSouth){
                        turnleft();
                    }
                    while(nextToABeeper){
                        pickbeeper();
                    }
                    if(frontIsClear){
                        move();
                    }else{
                        while(!facingNorth){
                            turnleft();
                        }
                        if(facingNorth && frontIsClear){
                            move();
                        }
                    }
                }else{
                    while(!facingWest){
                        turnleft();
                    }
                    while(frontIsClear){
                        move();
                    }
                    if(rightIsBlocked){
                        while(!facingSouth){
                            turnleft();
                        }
                    }
                    while(frontIsClear){
                        move();
                    }
                    while(anyBeepersInBeeperBag){
                        putbeeper();
                    }
                    while(!facingWest){
                        turnleft();
                    }
                }
            }
        }
        turnoff();
    }
}
